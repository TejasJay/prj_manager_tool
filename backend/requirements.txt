fastapi==0.104.1
# FastAPI: Modern, fast web framework for building APIs
# - Automatic API documentation (Swagger/OpenAPI)
# - Type hints and validation
# - Async support for better performance
# - This is our main framework for the backend

uvicorn[standard]==0.24.0
# Uvicorn: ASGI server to run FastAPI
# - The server that actually runs our FastAPI application
# - [standard] includes extra dependencies for better performance
# - Handles HTTP requests and responses

python-multipart==0.0.9
# python-multipart: Form data parsing for FastAPI
# - Required for handling form data (OAuth2PasswordRequestForm)
# - Needed for login endpoint that uses form data
# - Parses multipart/form-data and application/x-www-form-urlencoded
# - Updated from 0.0.6 to 0.0.9 for better compatibility

# ============================================
# Database & ORM
# ============================================
sqlalchemy==2.0.23
# SQLAlchemy: Python SQL toolkit and Object-Relational Mapping (ORM)
# - Allows us to interact with PostgreSQL using Python objects
# - Instead of writing raw SQL, we write Python code
# - Handles database connections, queries, and relationships

psycopg2-binary==2.9.9
# psycopg2: PostgreSQL adapter for Python
# - The actual driver that connects Python to PostgreSQL
# - Binary version is easier to install (no compilation needed)
# - SQLAlchemy uses this under the hood to talk to PostgreSQL

alembic==1.12.1
# Alembic: Database migration tool
# - Manages changes to your database schema over time
# - Creates "migrations" (version control for your database)
# - Allows you to update database structure without losing data
# - Essential for managing database changes as your app evolves

# Authentication & Security
# ============================================
python-jose[cryptography]==3.3.0
# python-jose: JWT (JSON Web Token) library
# - Creates and verifies JWT tokens for user authentication
# - When users log in, we create a token they use for subsequent requests
# - [cryptography] adds encryption support

bcrypt==5.0.0
# bcrypt: Password hashing library
# - NEVER store passwords in plain text!
# - Hashes passwords using bcrypt algorithm
# - When user registers, we hash their password before storing
# - When they login, we verify the hash matches
# - We use bcrypt directly instead of passlib for better compatibility

# passlib[bcrypt]==1.7.4
# passlib: Password hashing library (DEPRECATED - using bcrypt directly)
# - Commented out due to compatibility issues with bcrypt 5.0.0
# - We now use bcrypt directly in app/auth.py

python-dotenv==1.0.0
# python-dotenv: Loads environment variables from .env file

# Data Validation & Settings
# ============================================
pydantic==2.5.0
# Pydantic: Data validation using Python type annotations
# - FastAPI uses this to validate request/response data
# - Ensures data types are correct (e.g., email is valid, age is a number)
# - Automatically converts and validates JSON data

email-validator==2.2.0
# email-validator: Email validation for Pydantic
# - Required when using EmailStr type in Pydantic schemas
# - Validates email format (e.g., user@example.com)
# - Used in UserCreate and UserUpdate schemas
# - Updated from 2.1.0 (yanked) to 2.2.0

pydantic-settings==2.1.0
# pydantic-settings: Settings management using Pydantic
# - Extends Pydantic for managing application settings
# - Used in our config.py to load and validate environment variables
# - Type-safe configuration management
